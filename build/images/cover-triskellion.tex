% Background
\fill[coverbg] (-2,-2) rectangle (2,2);

% Triple yin-yang triskelion (mitsudomoe)
% Three interlocking comma-shaped arms in track colors
% Outer circle R=1.5, inner circles r=0.75
% Inner circle centers at 90/210/330 degrees, distance r from origin
% Each arm: 120-degree outer arc + two semicircular inner arcs through center

\foreach \col/\rot in {trackone/0, tracktwo/120, trackthree/240} {
  \begin{scope}[rotate=\rot]
    \fill[\col] (0,1.5)
      arc(90:-30:1.5)       % outer circle, 120 degrees clockwise
      arc(330:150:0.75)     % inner circle (concave edge), to origin
      arc(-90:90:0.75)      % inner circle (convex edge), back to top
      -- cycle;
  \end{scope}
}

\end{tikzpicture}
