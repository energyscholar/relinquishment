% --- Full-height track stripe + rotated label (right margin) ---
% No stripe on pages before first \settrack call (front matter, TOC, etc.)
\AddToShipoutPictureBG{%
  \ifx\currenttracklabel\empty\else
    \colorlet{@trackstripe}{\currenttrackcolor}%
    \colorlet{@tracklabel}{\currenttrackcolor!40!white}%
    % Colored stripe — 8pt wide, full page height, right edge
    \AtPageUpperLeft{%
      \put(\LenToUnit{\paperwidth-10pt},\LenToUnit{-\paperheight}){%
        \textcolor{@trackstripe}{\rule{8pt}{\paperheight}}%
      }%
    }%
    % Rotated track label — centered vertically, right margin
    \AtPageUpperLeft{%
      \put(\LenToUnit{\paperwidth-22pt},\LenToUnit{-0.5\paperheight}){%
        \rotatebox[origin=c]{90}{\textcolor{@tracklabel}{\sffamily\footnotesize\currenttracklabel}}%
      }%
    }%
  \fi
}
